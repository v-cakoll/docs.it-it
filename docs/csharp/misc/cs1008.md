 L'esempio seguente genera l'errore CS1008:  
  
```  
// CS1008.cs  
abstract public class clx  
{  
   enum splitch : char   // CS1008, char not valid type for enums  
   {  
      x, y, z  
   }  
  
   public static void Main()  
   {  
   }  
}  
```
