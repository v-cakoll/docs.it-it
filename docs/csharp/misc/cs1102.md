## <a name="example"></a><span data-ttu-id="ef758-108">Esempio</span><span class="sxs-lookup"><span data-stu-id="ef758-108">Example</span></span>

public static class Extensions  
{  
    // No type parameters.  
        public static void Test(this out int i) {} // CS1102

}
