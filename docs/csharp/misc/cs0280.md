## <a name="example"></a><span data-ttu-id="a71b8-112">Esempio</span><span class="sxs-lookup"><span data-stu-id="a71b8-112">Example</span></span>  
 <span data-ttu-id="a71b8-113">L'esempio seguente genera l'errore CS0280.</span><span class="sxs-lookup"><span data-stu-id="a71b8-113">The following sample generates CS0280.</span></span>  
  
```  
// CS0280.cs  
using System;  
using System.Collections;  
  
public class ValidBase: IEnumerable  
{  
   IEnumerator IEnumerable.GetEnumerator()  
   {  
      yield return 0;  
   }  
  
   internal IEnumerator GetEnumerator()  
   {  
      yield return 0;  
   }  
}  
  
class Derived : ValidBase  
{  
   // field, not method  
   new public int GetEnumerator;  
}  
  
public class Test  
{  
   public static void Main()  
   {  
      foreach (int i in new Derived()) {}   // CS0280  
   }  
}  
```
